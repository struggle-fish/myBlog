## 基本使用


```py

# int（整数）
a = 100
b = -9999999999999999999   # 支持极大数值
c = 2 ** 100               # 任意精度

# float（浮点数）
x = 3.14159
y = 2.0                    # 带小数点就是 float
z = 1.5e8                  # 科学计数法：150000000.0
w = -0.000123

# complex（复数）
comp = 3 + 4j
print(comp.real)           # 3.0
print(comp.imag)           # 4.0
print(comp.conjugate())    # (3-4j)

```



## 类型转换

```py

# 字符串 → 数字（最常用）
print(int("123"))          # 123
print(float("3.14"))       # 3.14
print(complex("3+4j"))     # (3+4j)

# 数字之间转换
print(float(100))          # 100.0
print(int(3.9))            # 3 （直接截断小数部分）
print(int(-3.9))           # -3
print(complex(5))          # (5+0j)

# 安全转换（推荐写法，避免程序崩溃）
def safe_int(value):
    try:
        return int(value)
    except (ValueError, TypeError):
        return 0

# bool 与 int 互相转换
print(int(True))           # 1
print(int(False))          # 0
print(bool(5))             # True
print(bool(0))             # False

```


## 常用API

### 通用数字操作

```py

num1 = -15.67
num2 = 10
num3 = 3.14159

print(abs(num1))                    # 绝对值 → 15.67
print(max(5, 10, 3))                # 最大值 → 10
print(min([1, 5, 9]))               # 最小值 → 1
print(pow(2, 10))                   # 幂运算 → 1024
print(pow(2, 10, 7))                # (2**10) % 7 → 2
print(round(num3, 2))               # 四舍五入保留2位小数 → 3.14

```



### 数学模块 math


```py

import math

print(math.pi)                      # 圆周率
print(math.e)                       # 自然常数
print(math.sqrt(16))                # 开平方 → 4.0
print(math.ceil(3.2))               # 向上取整 → 4
print(math.floor(3.9))              # 向下取整 → 3
print(math.fabs(-5.5))              # 浮点绝对值
print(math.sin(math.pi / 2))        # 正弦函数

```

### 随机数模块

```py

import random

print(random.random())              # 0.0 ~ 1.0 之间的随机浮点数
print(random.randint(1, 100))       # 1~100 之间的随机整数
print(random.uniform(1.5, 10.5))    # 指定范围内的随机浮点数
print(random.choice([10, 20, 30]))  # 从列表中随机选一个

```


###  decimal 模块


```py

from decimal import Decimal, getcontext

getcontext().prec = 28              # 设置精度

price = Decimal('19.99')            # 必须用字符串初始化！
quantity = Decimal('3')
tax = Decimal('0.08')

total = price * quantity * (1 + tax)
print(total)                        # 64.7676 （精确，不会出现 float 的误差）
print(total.quantize(Decimal('0.01')))  # 四舍五入到分 → 64.77


```
