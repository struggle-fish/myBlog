```py
def get_elem(self, i: int) -> int:
    """
    按位查找操作：获取第i个位置的元素
    :param i: 位序（从1开始）
    :return: 第i个位置的元素值
    """
    # 健壮性判断：位序必须合法
    if i < 1 or i > self.length:
        raise ValueError(f"查找位置不合法！范围：1~{self.length}")
    # 位序i → 数组下标i-1
    return self.data[i - 1]

```

## 按值查找




![LOGO](/public/image/datastructure/ScreenShot_2026-04-01_131154_412.png)


:::code-group

```py
def locate_elem(self, e: int) -> int:
    """
    按值查找操作：找到第一个值等于e的元素，返回其位序（从1开始）
    :param e: 要查找的元素值
    :return: 位序（1~length），查找失败返回0
    """
    # 遍历数组，从下标0到length-1
    for i in range(self.length):
      # 找到值相等的元素，返回位序i+1
      if self.data[i] == e:
        return i + 1
    # 遍历完没找到，返回0
    return 0

```


```js
/**
 * 按值查找操作：找到第一个值等于e的元素，返回其位序（从1开始）
 * @param {number} e - 要查找的元素值
 * @returns {number} 位序（1~length），查找失败返回0
 */
locateElem(e) {
  // 遍历数组，从下标0到length-1
  for (let i = 0; i < this.length; i++) {
    // 找到值相等的元素，返回位序i+1
    if (this.data[i] === e) {
      return i + 1;
    }
  }
  // 遍历完没找到，返回0
  return 0;
}

```
