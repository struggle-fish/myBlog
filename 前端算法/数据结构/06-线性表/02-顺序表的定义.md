# 顺序表的定义



![LOGO](/public/image/datastructure/ScreenShot_2026-03-30_100920_190.png)
## 简介

顺序表 —— 用**顺序存储**的方式实现线性表顺序存储。

把逻辑上相邻的元素存储在物理位置上也相邻的存储单元中，元素之间的关系由存储单元的邻接关系来体现。

```js [js版本]

class SqList {
  static MaxSize = 10;
  data = [];
  length = 0;

  constructor() {
    this.data = new Array(SqList.MaxSize);

      this.data[i] = 0;
    }
    this.length = 0;
  }

  toString() {
    return `[${this.data.join(', ')}]`;
  }
```

```java [java版本]

public class SqList {
  // 最大长度
  private static final int MaxSize = 10;
  // 存放数据元素（本例中为 int 类型）
  private int[] data;
  // 当前长度
  private int length;

  // 构造方法 = InitList 初始化函数
  public SqList() {
    this.data = new int[MaxSize];
    // 将所有数据元素设置为默认初始值 0
    for (int i = 0; i < MaxSize; i++) {
      this.data[i] = 0;
    }
    // 初始长度为 0
    this.length = 0;
  }

  @Override
  public String toString() {
    StringBuilder sb = new StringBuilder();
    sb.append("[");
    for (int i = 0; i < MaxSize; i++) {
      sb.append(data[i]);
      if (i < MaxSize - 1) {
        sb.append(", ");
      }
    }
    sb.append("]");
    return sb.toString();
  }

  // 测试入口 = main 函数
  public static void main(String[] args) {
    // 声明并初始化顺序表
    SqList L = new SqList();
    System.out.println(L);
    // 后续操作...
  }
}
```

## 动态分配



![LOGO](/public/image/datastructure/ScreenShot_2026-03-27_232903_330.png)

![LOGO](/public/image/datastructure/ScreenShot_2026-03-27_233246_710.png)



:::code-group

```py [python版]
class SeqList:
    # 类属性：初始容量（对应C语言的InitSize）
    INIT_SIZE = 10

    def __init__(self):
        # 初始化动态数组（对应malloc申请空间）
        self.data = [0] * SeqList.INIT_SIZE  # 用列表模拟动态数组
        self.max_size = SeqList.INIT_SIZE  # 最大容量
        self.length = 0  # 当前有效元素个数

    def increase_size(self, len_val):
        """
        动态扩容方法（对应C语言的IncreaseSize）
        :param len_val: 额外增加的容量大小
        """
        # 暂存原数组（对应int *p = L.data;）
        old_data = self.data
        # 申请新的更大空间
        self.data = [0] * (self.max_size + len_val)
        # 复制原数据到新数组
        for i in range(self.length):
            self.data[i] = old_data[i]
        # 更新最大容量
        self.max_size += len_val
        # 原列表由Python自动回收（对应free(p)）

    def print(self):
        """辅助打印方法"""
        print("当前元素：", end="")
        print(self.data[:self.length])  # 只打印有效元素
        print(f"当前长度：{self.length}")
        print(f"最大容量：{self.max_size}")


if __name__ == "__main__":
    # 声明并初始化顺序表
    L = SeqList()
    print("=== 初始化后 ===")
    L.print()

    # 模拟插入几个元素
    L.data[0] = 10
    L.data[1] = 20
    L.data[2] = 30
    L.length = 3
    print("\n=== 插入3个元素后 ===")
    L.print()

    # 扩容5个位置
    L.increase_size(5)
    print("\n=== 扩容5个位置后 ===")
    L.print()

```

```js [js版]
class SeqList {
    // 静态属性：初始容量（对应C语言的InitSize）
    static INIT_SIZE = 10;

    constructor() {
        // 初始化动态数组（对应malloc申请空间）
        this.data = new Array(SeqList.INIT_SIZE).fill(0);
        this.maxSize = SeqList.INIT_SIZE;  // 最大容量
        this.length = 0;  // 当前有效元素个数
    }

    // 动态扩容方法（对应C语言的IncreaseSize）
    increaseSize(len) {
        // 暂存原数组（对应int *p = L.data;）
        const oldData = this.data;
        // 申请新的更大空间
        this.data = new Array(this.maxSize + len).fill(0);
        // 复制原数据到新数组
        for (let i = 0; i < this.length; i++) {
            this.data[i] = oldData[i];
        }
        // 更新最大容量
        this.maxSize += len;
        // 原数组由JS自动垃圾回收（对应free(p)）
    }

    // 辅助打印方法
    print() {
        console.log("当前元素：", this.data.slice(0, this.length));
        console.log(`当前长度：${this.length}`);
        console.log(`最大容量：${this.maxSize}`);
    }
}

// 测试入口 = main函数
{
    // 声明并初始化顺序表
    const L = new SeqList();
```

```java [java版]
public class SeqList {
    // 初始容量（对应C语言的InitSize）
    private static final int INIT_SIZE = 10;
    // 指向动态数组的引用（对应C语言的ElemType *data）
    private int[] data;
    // 最大容量（对应C语言的MaxSize）
    private int maxSize;
    // 当前有效元素个数（对应C语言的length）
    private int length;

    // 构造方法 = InitList 初始化函数
    public SeqList() {
        // 申请初始容量的数组空间（对应malloc）
        this.data = new int[INIT_SIZE];
        this.length = 0;
        this.maxSize = INIT_SIZE;
    }

    // 动态扩容方法 = IncreaseSize 函数
    // len: 需要额外增加的容量大小
    public void increaseSize(int len) {
        // 暂存原数组引用（对应int *p = L.data;）
        int[] oldData = this.data;
        // 申请新的更大空间（对应malloc新数组）
        this.data = new int[this.maxSize + len];
        // 复制原数据到新数组（对应for循环拷贝）
        for (int i = 0; i < this.length; i++) {
            this.data[i] = oldData[i];
        }
        // 更新最大容量
        this.maxSize += len;
        // 原数组空间由GC自动回收（对应free(p)）
    }

    // 辅助打印方法，方便查看状态
    public void print() {
        System.out.print("当前元素：[");
        for (int i = 0; i < this.length; i++) {
            System.out.print(this.data[i]);
            if (i != this.length - 1) System.out.print(", ");
        }
        System.out.println("]");
        System.out.println("当前长度：" + this.length);
        System.out.println("最大容量：" + this.maxSize);
    }

    // 测试入口 = main函数
    public static void main(String[] args) {
        // 声明并初始化顺序表
        SeqList L = new SeqList();
        System.out.println("=== 初始化后 ===");
        L.print();

        // 模拟插入几个元素（这里先手动赋值演示）
        L.data[0] = 10;
        L.data[1] = 20;
        L.data[2] = 30;
        L.length = 3;
        System.out.println("\n=== 插入3个元素后 ===");
        L.print();

        // 扩容5个位置
        L.increaseSize(5);
        System.out.println("\n=== 扩容5个位置后 ===");
        L.print();
    }
}
```
