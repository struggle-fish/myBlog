```py
def list_insert(self, i: int, e: int) -> bool:
    """
    顺序表插入操作（带健壮性判断，返回是否成功）
    :param i: 位序（从1开始）
    :param e: 要插入的元素
    :return: True-插入成功，False-插入失败
    """
    # 1. 判断插入位置是否合法
    if i < 1 or i > self.length + 1:
        return False
    # 2. 判断存储空间是否已满
    if self.length >= SqList.MaxSize:
        return False

    # 3. 将第i个元素及之后的元素后移
    for j in range(self.length, i - 1, -1):
        self.data[j] = self.data[j - 1]

    # 4. 在位置i处放入e
    self.data[i - 1] = e

    # 5. 长度加1
    self.length += 1

    # 6. 插入成功
    return True
```

```js
/**
 * 顺序表插入操作（带健壮性判断，返回是否成功）
 * @param {number} i - 位序（从1开始）
 * @param {number} e - 要插入的元素
 * @returns {boolean} true-插入成功，false-插入失败
 */
listInsert(i, e) {
    // 1. 判断插入位置是否合法
    if (i < 1 || i > this.length + 1) {
        return false;
    }
    // 2. 判断存储空间是否已满
    if (this.length >= SqList.MaxSize) {
        return false;
    }

    // 3. 将第i个元素及之后的元素后移
    for (let j = this.length; j >= i; j--) {
        this.data[j] = this.data[j - 1];
    }

    // 4. 在位置i处放入e
    this.data[i - 1] = e;

    // 5. 长度加1
    this.length++;

    // 6. 插入成功
    return true;
}

```


```java


/**
 * 顺序表插入操作（带健壮性判断，返回是否成功）
 * @param i 位序（从1开始）
 * @param e 要插入的元素
 * @return true-插入成功，false-插入失败
 */
public boolean listInsert(int i, int e) {
    // 1. 判断插入位置是否合法（对应 if(i<1 || i>L.length+1)）
    if (i < 1 || i > this.length + 1) {
        return false;
    }
    // 2. 判断存储空间是否已满（对应 if(L.length>=MaxSize)）
    if (this.length >= MaxSize) {
        return false;
    }

    // 3. 将第i个元素及之后的元素后移
    for (int j = this.length; j >= i; j--) {
        this.data[j] = this.data[j - 1];
    }

    // 4. 在位置i处放入e
    this.data[i - 1] = e;

    // 5. 长度加1
    this.length++;

    // 6. 插入成功（对应 return true;）
    return true;
}
```


:::


原理：

- 把插入位置后面的所有元素，全部往后挪一位！

- `j` 就是从最后一个开始，指挥他们一个个往后退！

- `j--` 就是指挥完一个，往前找下一个！

- 循环做的事：把元素一个个往后复制，腾出插入的坑



::: tip

Python 的 range 三个参数是什么意思？

range(start, stop, step)

- start：从哪里开始
- stop：到哪里停止（不包含）
- step：每次加多少（-1 就是倒着走）
:::

```py
def list_delete(self, i: int) -> int:
    """
    顺序表删除操作
    :param i: 要删除元素的位序（从1开始）
    :return: 被删除的元素值
    """
    # 1. 判断位序是否合法
    if i < 1 or i > self.length:
      raise ValueError(f"删除位置不合法！范围：1~{self.length}")

    # 2. 保存被删除的元素
    e = self.data[i - 1]

    # 3. 将第i个位置后的元素前移
    for j in range(i, self.length):
      self.data[j - 1] = self.data[j]

    # 4. 长度减1
    self.length -= 1

    # 5. 返回被删除的元素
    return e

```


```js
/**
 * 顺序表删除操作
 * @param {number} i - 要删除元素的位序（从1开始）
 * @returns {number} 被删除的元素值
 */
listDelete(i) {
  // 1. 判断位序是否合法
  if (i < 1 || i > this.length) {
      throw new Error(`删除位置不合法！范围：1~${this.length}`);
  }

  // 2. 保存被删除的元素
  const e = this.data[i - 1];

  // 3. 将第i个位置后的元素前移
  for (let j = i; j < this.length; j++) {
    this.data[j - 1] = this.data[j];
  }

  // 4. 长度减1
  this.length--;

  // 5. 返回被删除的元素
  return e;
}

```

```java
/**
 * 顺序表删除操作
 * @param i 要删除元素的位序（从1开始）
 * @return 被删除的元素值；删除失败则抛出异常
 */
public int listDelete(int i) {
    // 1. 判断位序是否合法（对应 if(i<1 || i>L.length)）
    if (i < 1 || i > this.length) {
      throw new IllegalArgumentException("删除位置不合法！范围：1~" + this.length);
    }

    // 2. 保存被删除的元素（对应 e = L.data[i-1]）
    int e = this.data[i - 1];

    // 3. 将第i个位置后的元素前移（对应 for(int j=i; j<L.length; j++)）
    for (int j = i; j < this.length; j++) {
      this.data[j - 1] = this.data[j];
    }

    // 4. 长度减1（对应 L.length--）
    this.length--;

    // 5. 返回被删除的元素
    return e;
}
```


:::


插入是元素后移，删除是元素前移。也就是，从当前元素往后遍历，把后面的元素往前移动，即 `j++`
